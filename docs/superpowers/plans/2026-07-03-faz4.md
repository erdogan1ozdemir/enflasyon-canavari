```ts
export type ParaBirimi = "TL" | "usd" | "gram-altin";
export function paraToUrun(prices: PricePoint[], tutar: number, birim: ParaBirimi, urunId: string, yil: number): number | null {
  if (tutar < 0) return null;
  let tutarTL = tutar;
  if (birim !== "TL") {
    const kur = fiyatBul(prices, birim, yil);
    if (kur === null) return null;
    tutarTL = tutar * kur;
  }
  return kacXEder(prices, tutarTL, urunId, yil);
}
export function urunToPara(prices: PricePoint[], miktar: number, urunId: string, yil: number): { tl: number; usd: number | null } | null {
  if (miktar < 0) return null;
  const f = fiyatBul(prices, urunId, yil);
  if (f === null) return null;
  const tl = miktar * f;
  const kur = fiyatBul(prices, "usd", yil);
  return { tl, usd: kur !== null && kur !== 0 ? tl / kur : null };
}
```
(`fiyatBul`, `kacXEder` importlarını `@ec/data`'dan mevcut import satırına ekle.)
`web/lib/profile.ts`:
```ts
export interface Profil { adSoyad?: string; maas?: number }
const KEY = "ec_profil";
export function getProfil(): Profil {
  if (typeof window === "undefined") return {};
  try { return JSON.parse(window.localStorage.getItem(KEY) ?? "{}") as Profil; }
  catch { return {}; }
}
export function setProfil(p: Profil): void {
  if (typeof window === "undefined") return;
  window.localStorage.setItem(KEY, JSON.stringify(p));
}
```
